```c
int Min(int n, int m) {
    int i, j, k, temp, Min;
    state[0][1] = 0;
    for (i = 1; i <= n; i++) {
        state[i][1] = state[i - 1][1] + a[i];
    }
    for (i = 2; i <= n; i++) {
        for (j = 2; j <= m; j++) {
            temp = 1000000;
            for (k = 1; k < i; k++) {
                Min = state[i][1] - state[k][1] > state[k][j - 1] ? (state[i][1] - state[k][1]) : state[k][j - 1];
                if (temp > Min) temp = Min;
            }
            state[i][j] = temp;
        }
    }
    return state[n][m];
}
```

那么可得时间复杂度为O(mni);
